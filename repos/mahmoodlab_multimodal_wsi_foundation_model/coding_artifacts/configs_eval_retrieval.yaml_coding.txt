## Code: configs/eval/retrieval.yaml
```python
# configs/eval/retrieval.yaml
# Retrieval evaluation configuration for TITAN-style slide embeddings.
# This file is config-only and binds to values provided in config.yaml.

eval:
  name: retrieval
  enabled: true
  task_type: retrieval

  runtime:
    python_version: "3.9.16"
    pytorch_version: "2.0.1"
    cuda_version: "11.8"
    deterministic: true

  hardware:
    gpus: 1
    gpu_type: "NVIDIA 3090 24GB"
    num_workers: 4
    pin_memory: true

  data:
    patch_size_px: 512
    magnification: "20x"
    patch_feature_dim: 768

    artifacts:
      features_filename: "features.h5"
      grid_filename: "grid.pt"
      pairs_filename: "pairs.jsonl"
      splits_filename: "splits.csv"

    # Retrieval follows fixed query/database definitions from split metadata.
    split_policy:
      # For unimodal slide retrieval: use train+val as database and test as query
      # unless dataset-specific manifests provide explicit support/query partitions.
      unimodal_database_splits: ["train", "val"]
      unimodal_query_split: "test"

      # For cross-modal retrieval, evaluate on fixed paired split.
      cross_modal_eval_split: "test"
      use_validation_split_for_model_selection: false

  model:
    encoder_checkpoint: "titan_final.ckpt"
    freeze_backbone: true

    slide_encoder:
      architecture: "ViT in feature space"
      num_layers: 6
      num_attention_heads: 12
      head_dim: 64
      embedding_dim: 768
      mlp_hidden_dim: 3072
      positional_encoding: "2D ALiBi (Euclidean-distance based bias)"

    multimodal:
      framework: "CoCa"
      reconstruction_queries: 128
      text_encoder_source: "CONCHv1.5 pretrained text encoder"
      text_decoder_source: "CONCHv1.5 pretrained multimodal decoder"
      text_encoder_layers: 12
      text_decoder_layers: 12
      text_embedding_dim: 768
      text_hidden_dim: 3072

  protocol:
    unimodal_slide_retrieval:
      enabled: true

      # Paper method: evaluate nearest neighbors at K={1,3,5}.
      k_values: [1, 3, 5]

      # Distance is computed after centering and L2 normalization.
      distance: "euclidean"

      # Paper method reports MVAcc@5 in addition to Acc@K.
      majority_vote_k: 5

      # Deterministic tie handling for MVAcc@5 majority voting.
      majority_vote_tie_break: "lowest_distance_sum"

    cross_modal_retrieval:
      enabled: true

      # Paper method: Recall@K at K={1,3,5,10} and mean recall.
      recall_k_values: [1, 3, 5, 10]
      similarity: "cosine"
      report_mean_recall: true

      # Evaluate both directions in aligned embedding space.
      directions:
        - "slide_to_report"
        - "report_to_slide"

      # Success criterion is class-consistent retrieval.
      class_based_success: true

  preprocessing:
    # Unimodal retrieval preprocessing per paper methods section.
    unimodal:
      center_embeddings: true
      l2_normalize_embeddings: true
      fit_center_on_database_only: true

    # Cross-modal retrieval preprocessing per paper methods section.
    cross_modal:
      center_slide_embeddings: false
      center_text_embeddings: false
      l2_normalize_slide_embeddings: true
      l2_normalize_text_embeddings: true

  metrics:
    unimodal:
      - "acc_at_1"
      - "acc_at_3"
      - "acc_at_5"
      - "mvacc_at_5"

    cross_modal:
      - "recall_at_1"
      - "recall_at_3"
      - "recall_at_5"
      - "recall_at_10"
      - "mean_recall"

    aggregate:
      bootstrap_samples: 1000
      report_mean_std: true
      track_per_fold: true

  io:
    output_dir: "./outputs/eval/retrieval"
    unimodal_predictions_filename: "retrieval_unimodal_rankings.csv"
    cross_modal_predictions_filename: "retrieval_cross_modal_rankings.csv"
    summary_filename: "retrieval_summary.csv"
    metrics_filename: "retrieval_metrics.json"
    write_json: true
    write_csv: true

  constraints:
    require_embedding_dim_match: true
    required_embedding_dim: 768
    require_frozen_model: true
    require_label_alignment_between_query_and_database: true
    require_pair_alignment_for_cross_modal: true
    require_deterministic_eval: true
```
